% Macros
% Two column slide - useful to create a slide with two columns or two figures on a slide
%   Example: the command below creates two columns. The left column occupies 30% of the
%    slide width. The text: "example text for column one" is placed in the left column
%    and the text: "example text for column two" is placed in the right column
%    \twocols[0.30]{example text for column one}{example text for column two}
%\vspace{-\topsep}\vspace{2pt}
\newcommand{\twocols}[3][0.5]{%
	\vspace{-\baselineskip}\vspace{2pt}%
	\begin{columns}[T]%
		\begin{column}{#1\columnwidth}#2\end{column}%
		\begin{column}{\columnwidth-{#1}\columnwidth}#3\end{column}%
	\end{columns}%
}
% Fix left-align on bullets
% \vspace{-\topsep}\vspace{2pt}
\newcommand{\twocolsfix}[3][0.5]{%
	\vspace{-\baselineskip}\vspace{2pt}%
	\begin{columns}%
		\begin{column}{#1\columnwidth-0.015\columnwidth}#2\end{column}%
		\begin{column}{\columnwidth-{#1}\columnwidth-0.015\columnwidth}#3\end{column}%
	\end{columns}%
}
